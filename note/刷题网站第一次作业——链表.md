## 1. 合并两个有序链表

```text
分值: 10 分
描述
将两个升序链表合并为一个新的升序链表并返回。新链表是通过拼接给定的两个链表的所有节点组成的。
输入
输入两个顺序表l1，l2，l1和l2均按非递减顺序排列，两个链表的节点数目范围是 [0, 50]，-100 <= Node.val <= 100
-1表示链表结束符
输出
输出合并后的顺序表
输入样例

1 2 4 -1
1 3 4 -1
输出样例

1 1 2 3 4 4
```

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Node{
    int data;
    struct Node* next;
} Node;

Node* createList(){
    Node* head = NULL;
    Node* trail = NULL;

    int i;
    while(scanf("%d", &i) && i != -1){
        Node* newnode = (Node*)malloc(sizeof(Node));
        newnode->data = i;
        newnode->next = NULL;

        if(head == NULL){
            head = trail = newnode;
        } else {
            trail->next = newnode;
            trail = newnode;
        }
    }
    return head;
}

Node* merge(Node* L1, Node* L2){
    Node L3;
    L3.next = NULL;
    Node* p = &L3;

    while(L1 && L2){
        if(L1->data <= L2->data){
            p->next = L1;
            L1 = L1->next;
        } else {
            p->next = L2;
            L2 = L2->next;
        }
        p = p->next;
    }

    p->next = L1 ? L1 : L2;

    return L3.next;
}

int main(){
    Node* L1 = createList();
    Node* L2 = createList();

    Node* L3 = merge(L1, L2);

    Node* p = L3;
    while(p){
        printf("%d ", p->data);
        p = p->next;
    }

    return 0;
}
```

## 5. 合并两个链表

```text
分值: 10 分
描述
给你两个链表 list1 和 list2 ，它们包含的元素分别为 n 个和 m 个。
请你将 list1 中下标从 a 到 b 的全部节点都删除，并将list2 接在被删除节点的位置。
```

```c
#include <stdlib.h>

    int data;
    struct Node* next;
} Node;

    Node* head = NULL;
    Node* tail = NULL;

    dummy.next = L1;

    Node* p1 = &dummy;
    int i = 0;

    // 找到 a-1 位置

        p1 = p1->next;
        i++;
```

## 8. 反转链表Ⅱ

```text
分值: 10 分
描述
给你单链表的头指针 head 和两个整数 left 和 right ，其中 left <= right 。请你反转从位置 left 到位置 right 的链表节点，返回 反转后的链表 。
输入
链表中节点数目为 n
1 <= n <= 500
-500 <= Node.val <= 500
1 <= left <= right <= n
-1表示链表结束符
输出
输出链表
输入样例

1 2 3 4 5 -1 2 4
输出样例

1 4 3 2 5
```

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Node{
    int data;
    struct Node* next;
}Node;

// 创建链表（尾插法）
Node* createL(){
    Node* head = NULL;
    Node* tail = NULL;
    int x;

    while(scanf("%d", &x) && x != -1){
        Node* node = (Node*)malloc(sizeof(Node));
        node->data = x;
        node->next = NULL;

        if(head == NULL){
            head = tail = node;
        }else{
            tail->next = node;
            tail = node;
        }
    }
    return head;
}

// 反转区间 [a, b]
Node* inverseL(Node* L, int a, int b){

    Node dummy;
    dummy.next = L;

    Node* p1 = &dummy;

    // 1️⃣ 找到 a-1 位置
    for(int i = 1; i < a; i++){
        p1 = p1->next;
    }

    // 2️⃣ 从 a 开始反转
    Node* cur = p1->next;   // a
    Node* pre = NULL;
    Node* nex = NULL;

    // 反转 b-a+1 个节点
    for(int i = 0; i <= b - a; i++){
        nex = cur->next;
        cur->next = pre;
        pre = cur;
        cur = nex;
    }

    // 3️⃣ 拼接
    Node* p2 = p1->next;   // 原来的 a（现在是尾）
    p1->next = pre;        // 接上反转后的头
    p2->next = cur;        // 接上后半部分

    return dummy.next;
}

// 输出链表
void printList(Node* head){
    while(head){
        printf("%d", head->data);
        if(head->next) printf(" ");
        head = head->next;
    }
}

    Node* L = createL();

    int a, b;
    scanf("%d %d", &a, &b);

    L = inverseL(L, a, b);

    printList(L);

    return 0;
}
```

## 9. 链表最大孪生和

```text
分值: 10 分
描述
在一个大小为 n 且 n 为 偶数 的链表中，对于 0 <= i <= (n / 2) - 1 的 i ，第 i 个节点（下标从 0 开始）的孪生节点为第 (n-1-i) 个节点 。
比方说，n = 4 那么节点 0 是节点 3 的孪生节点，节点 1 是节点 2 的孪生节点。这是长度为 n = 4 的链表中所有的孪生节点。
孪生和定义为一个节点和它孪生节点两者值之和。
给你一个长度为偶数的链表的头节点 head ，请你返回链表的 最大孪生和 。
输入
输入链表的节点数目是 [2, 10^5] 中的 偶数 。
1 <= Node.val <= 10^5
-1是链表结束符
输出
输出孪生和
输入样例

5 4 2 1 -1
输出样例

6
```

## 11. 重排链表

```text
分值: 10 分
描述
给定一个单链表 L 的头节点 head ，单链表 L 表示为：
L0 → L1 → … → Ln - 1 → Ln
请将其重新排列后变为：
L0 → Ln → L1 → Ln - 1 → L2 → Ln - 2 → …
不能只是单纯的改变节点内部的值，而是需要实际的进行节点交换。
输入
输入链表的长度范围为 [1, 5 * 10^4]
1 <= node.val <= 1000
-1表示链表结束符
输出
输出重排链表
输入样例

1 2 3 4 5 -1
输出样例

1 5 2 4 3
```

```c
#include<stdio.h>
#include<stdlib.h>
typedef struct Node{
    int data;
    struct Node*next;
}Node;
Node*createL(){
    Node*head=NULL;
    Node*trail=NULL;
    int i=0;
    while(scanf("%d",&i)&&i!=-1){
        Node*newNode=(Node*)malloc(sizeof(Node));
        newNode->data=i;
        newNode->next=NULL;
        if(head==NULL) head=trail=newNode;
        else{
            trail->next=newNode;
            trail=newNode;
        }
    }
    return head;
}
Node*inverse(Node*L){
    Node*pre=NULL;
    Node*cur=L;
    Node*nex=NULL;
    while(cur){
        nex=cur->next;
        cur->next=pre;
        pre=cur;
        cur=nex;
    }
    return pre;
}
Node*resort(Node*L1,Node*L2){
    Node*head=L1;
    Node*p1=L1;
    Node*p2=L1;
    Node*p3=L2;
    Node*p4=L2;
    while(p3&&p1){
        p1=p1->next;
        p3=p3->next;
        p4->next=p2->next;
        p2->next=p4;
        p2=p1;
        p4=p3;
    }
    return head;
}
int main(){
    Node*L1=createL();
    Node*slow=L1;
    Node*fast=L1;
    while(fast&&fast->next){
        slow=slow->next;
        fast=fast->next->next;
    }
    Node*L2=inverse(slow->next);
    slow->next=NULL;
    Node*L3=resort(L1,L2);
    while(L3){
        printf("%d",L3->data);
        if(L3->next) printf(" ");
        L3=L3->next;
    }
    return 0;
}
```

## 12. 删除链表的倒数第 N 个结点

```text
分值: 10 分
描述
给你一个链表，删除链表的倒数第 n 个结点，并且返回链表的头结点。
输入
输入链表和要删除的n ，链表中结点的数目为 sz，
1 <= sz <= 30，0 <= Node.val <= 100，1 <= n <= sz
-1表示链表结束符
输出
输出链表
输入样例

1 2 3 4 5 -1
2
输出样例

1 2 3 5
```

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct Node {
    int data;
    struct Node* next;
} Node;

// 创建链表（尾插法）
Node* createL() {
    Node *head = NULL, *tail = NULL;
    int x;
    while (scanf("%d", &x) && x != -1) {
        Node* newNode = (Node*)malloc(sizeof(Node));
        newNode->data = x;
        newNode->next = NULL;

        if (head == NULL) {
            head = tail = newNode;
        } else {
            tail->next = newNode;
            tail = newNode;
        }
    }
    return head;
}

// 删除倒数第n个节点
Node* removeNthFromEnd(Node* head, int n) {
    Node* dummy = (Node*)malloc(sizeof(Node));
    dummy->next = head;

    Node *fast = dummy, *slow = dummy;

    // fast先走n步
    for (int i = 0; i < n; i++) {
        fast = fast->next;
    }

    // 一起走
    while (fast->next != NULL) {
        fast = fast->next;
        slow = slow->next;
    }

    // 删除节点
    Node* temp = slow->next;
    slow->next = temp->next;
    free(temp);

    return dummy->next;
}

// 打印链表
void printL(Node* head) {
    while (head) {
        printf("%d ", head->data);
        head = head->next;
    }
}

int main() {
    Node* head = createL();
    int n;
    scanf("%d", &n);

    head = removeNthFromEnd(head, n);
    printL(head);

    return 0;
}
```
